      23     $fileExists = Test-Path $completionFile

      28         # Now check GitHub PR status
      29         Write-Host "Checking GitHub Actions status for PR #$PrNumber..."

      31
      32         $allPassed = $true
      33         $allFinished = $true
      34
      35         $checksObj = $checks | ConvertFrom-Json
      36         foreach ($check in $checksObj) {
      37             if ($check.status -ne "completed") { $allFinished = $false }
      38             if ($check.state -ne "success" -and $check.state -ne "neutral") { $allPassed = $false }
      39
      40             $color = if ($check.state -eq "success") { "Green" } elseif ($check.state -eq "failing") {       
         "Red" } else { "Yellow" }
      41             Write-Host "  - $($check.name): $($check.state) ($($check.status))" -ForegroundColor $color      
      42         }
      43
      44         if ($allFinished -and $allPassed) {
      45             Write-Host "`n[SUCCESS] Task '$TaskName' is complete and verified by CI." -ForegroundColor       
         Green
      46             exit 0
      47         } elseif ($allFinished -and -not $allPassed) {
      48             Write-Host "`n[FAILURE] CI failed for PR #$PrNumber. Task needs fix." -ForegroundColor Red       
      49             exit 1
      50         } else {
      51             Write-Host "Waiting for CI to finish..."
      52         }
      53     } else {
      54         Write-Host "." -NoNewline
      55     }
      56
